🎲 サイコロを振る！

import js from pyodide.ffi import create_proxy import random canvas = js.document.getElementById("dice-canvas") scene = js.THREE.Scene.new() camera = js.THREE.PerspectiveCamera.new(75, 1, 0.1, 1000) renderer = js.THREE.WebGLRenderer.new(js.Object.fromEntries([["canvas", canvas]])) renderer.setSize(400, 400) # サイコロの形状とマテリアル geometry = js.THREE.BoxGeometry.new(1, 1, 1) material = js.THREE.MeshNormalMaterial.new() cube = js.THREE.Mesh.new(geometry, material) scene.add(cube) camera.position.z = 3 # アニメーション処理 def animate_rotation(): for _ in range(60): cube.rotation.x += 0.1 cube.rotation.y += 0.1 renderer.render(scene, camera) # 出目を出す関数 def roll_dice(event): animate_rotation() result = random.randint(1, 6) js.document.getElementById("dice-result").innerText = f"出た目: {result}" # ボタンにイベント登録 js.document.getElementById("roll-button").addEventListener("click", create_proxy(roll_dice)) # 初期描画ループ def render_loop(*args): renderer.render(scene, camera) js.requestAnimationFrame(create_proxy(render_loop)) render_loop()
